    <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
    <script>
        let map;
        let userMarker;
        let markersGroup = [];
        let userLocation = [25.0408, 121.5160]; 
        let allStations = [];
        const API_URL = 'https://tcgbusfs.blob.core.windows.net/dotapp/youbike/v2/youbike_immediate.json';

        function initMap() {
            map = L.map('map', { zoomControl: false }).setView(userLocation, 15);
            L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);
            
            map.on('click', (e) => {
                updateUserMarker(e.latlng.lat, e.latlng.lng, "手動釘選位置");
            });
        }

        function updateUserMarker(lat, lng, label) {
            userLocation = [lat, lng];
            if (userMarker) {
                userMarker.setLatLng(userLocation);
            } else {
                userMarker = L.circleMarker(userLocation, {
                    radius: 12, fillColor: "#fab1a0", color: "#fff", weight: 3, fillOpacity: 0.9
                }).addTo(map);
            }
            userMarker.bindPopup(`<b>${label}</b>`).openPopup();
            updateNearbyUI();
            document.getElementById('location-status').innerHTML = `📍 目前位置：${label}`;
        }

        async function fetchData() {
            // 只有手動重新整理時顯示 loader，自動重新整理則靜默進行
            try {
                const response = await fetch(API_URL);
                allStations = await response.json();
                updateNearbyUI();
            } catch (error) {
                console.error("Fetch Error:", error);
            }
        }

        function updateNearbyUI() {
            if (!allStations.length) return;

            const stationsWithDist = allStations.map(s => ({
                ...s,
                distance: getDistance(userLocation[0], userLocation[1], s.latitude, s.longitude)
            }));

            const nearby20 = stationsWithDist.sort((a, b) => a.distance - b.distance).slice(0, 20);

            markersGroup.forEach(m => map.removeLayer(m));
            markersGroup = [];

            nearby20.forEach(s => {
                const m = L.circleMarker([s.latitude, s.longitude], {
                    radius: 8, fillColor: "#a29bfe", color: "#6c5ce7", weight: 2, fillOpacity: 0.7
                }).addTo(map);

                // 車位總數設定為 available_return_bikes + available_rent_bikes
                const totalCalculated = (parseInt(s.available_return_bikes) || 0) + (parseInt(s.available_rent_bikes) || 0);

                m.bindPopup(`
                    <b>${s.sna.replace('YouBike2.0_', '')}</b><br>
                    🚲 可借車輛: ${s.available_rent_bikes}<br>
                    🅿️ 車位總數: ${totalCalculated}<br>
                    🕒 ${s.updateTime}
                `);
                markersGroup.push(m);
            });

            renderCards(nearby20);
        }

        function renderCards(data) {
            const container = document.getElementById('card-list');
            container.innerHTML = '';
            data.forEach(s => {
                const card = document.createElement('div');
                card.className = 'card';
                const distText = s.distance < 1 ? `${(s.distance * 1000).toFixed(0)}m` : `${s.distance.toFixed(2)}km`;
                
                // 車位總數設定為 available_return_bikes + available_rent_bikes
                const totalCalculated = (parseInt(s.available_return_bikes) || 0) + (parseInt(s.available_rent_bikes) || 0);

                card.innerHTML = `
                    <div class="dist-badge">🏃 距離 ${distText}</div>
                    <div class="card-title">${s.sna.replace('YouBike2.0_', '')}</div>
                    <div class="card-info"><span>🚲 可借車輛</span><span class="badge">${s.available_rent_bikes}</span></div>
                    <div class="card-info"><span>🅿️ 車位總數</span><span>${totalCalculated}</span></div>
                    <div class="card-time">🕒 ${s.updateTime}</div>
                `;
                card.onclick = () => map.flyTo([s.latitude, s.longitude], 17);
                container.appendChild(card);
            });
        }

        function getDistance(lat1, lon1, lat2, lon2) {
            const R = 6371;
            const dLat = (lat2-lat1)*Math.PI/180;
            const dLon = (lon2-lon1)*Math.PI/180;
            const a = Math.sin(dLat/2)*Math.sin(dLat/2)+Math.cos(lat1*Math.PI/180)*Math.cos(lat2*Math.PI/180)*Math.sin(dLon/2)*Math.sin(dLon/2);
            return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
        }

        function refreshData() { 
            document.getElementById('loader').style.display = 'flex';
            fetchData().finally(() => {
                document.getElementById('loader').style.display = 'none';
            });
        }

        function centerToUser() { map.flyTo(userLocation, 16); }

        window.onload = () => {
            initMap();
            if (navigator.geolocation) {
                navigator.geolocation.getCurrentPosition(
                    p => updateUserMarker(p.coords.latitude, p.coords.longitude, "您的定位點"),
                    () => updateUserMarker(userLocation[0], userLocation[1], "預設點 (請手動點擊地圖)")
                );
            }
            fetchData();

            // 每 30 秒自動重新整理數據 (每 30000 毫秒)
            setInterval(() => {
                fetchData();
            }, 30000);
        };
    </script>
